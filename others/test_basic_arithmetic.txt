# Basic Arithmetic Test
# Tests: SET, ADD, CPY, HLT
Begin Data Section
0 0          # PC
1 500        # SP
2 0          # SYSCALL_RESULT
3 0          # INSTRUCTION_COUNT
50 0         # Test variable 1
51 0         # Test variable 2
52 0         # Test result
End Data Section

Begin Instruction Section
0 SET 10 50      # Set memory[50] = 10
1 SET 5 51       # Set memory[51] = 5
2 ADD 50 3       # Add 3 to memory[50], so memory[50] = 13
3 CPY 50 52      # Copy memory[50] to memory[52], so memory[52] = 13
4 ADDI 52 51     # Add memory[51] to memory[52], so memory[52] = 13 + 5 = 18
5 HLT            # Halt
End Instruction Section
